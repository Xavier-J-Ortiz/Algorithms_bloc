1) Calculating a tip -
def tip(cost)
  temp = cost
  temp = 2 * temp / 10
  return temp
end

2) Determine what to wear - 

def dress(temperature, precipitation)
  if weather == cold
    shirt = long_sleeve_shirt
    pants = jeans
    sweater = true
  else
    shirt = short_sleeve_shirt
    pants = slacks
    sweater = false
  end
  if precipitation == true
    boots = true
    hat = true
  else
    boots = false
    hat = false
  end
  return [shirt, pants, sweater, boots, hat]
end
